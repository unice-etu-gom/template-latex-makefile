\newcounter{todocounter}
\newcommand{\todonum}[2][]
{
	\stepcounter{todocounter}
	\todo[#1]{\thetodocounter: #2}
}
\newcommand{\todoGOM}[1]{\todonum[author=GOM,color={blue!20}]{#1}}
